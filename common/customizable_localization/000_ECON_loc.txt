internal_trade_scope_custom_loc_grain = {
    type = province

    # Customs union
    text = {
        localization_key = internal_trade_scope_customs_union_grain
        trigger = { # Check if the governorship is in a customs union
            governorship = { has_variable = federation_customs_union }
        }
    }
    # No customs union
    text = {
        localization_key = internal_trade_scope_country_grain
        trigger = { # Check if the governorship is in a customs union
            governorship = { NOT = { has_variable = federation_customs_union } }
        }
    }
}
