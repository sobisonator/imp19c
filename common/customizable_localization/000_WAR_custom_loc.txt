peace_province_selection_loc = {
	type = province
	text = {
		localization_key = loc_province_unselected
		trigger = {
			NOT = {
				owner = {
					any_in_list = {
						variable = list_of_wars_in_country
						is_target_in_variable_list = {
							name = provinces_available_from_both
							target = ROOT
						}
						any_war_participant = {
							any_in_list = {
								variable = war_scope_provinces
								is_target_in_variable_list = {
									name = peace_requested_provinces
									target = ROOT
								}
							}
						}
					}
				}
			}
		}
	}
	text = {
		localization_key = loc_province_selected
		trigger = {
			owner = {
				any_in_list = {
					variable = list_of_wars_in_country
					is_target_in_variable_list = {
						name = provinces_available_from_both
						target = ROOT
					}
					any_war_participant = {
						any_in_list = {
							variable = war_scope_provinces
							is_target_in_variable_list = {
								name = peace_requested_provinces
								target = ROOT
							}
						}
					}
				}
			}			
		}
	}
}

	type = province
	text = {
		localization_key = DIPLOMATIC_PLAY_WAR_WILLING
		trigger = {
			var:war_assessment > 0
		}
	}
	text = {
		localization_key = DIPLOMATIC_PLAY_WAR_CONSIDERING
		trigger = {
			var:war_assessment <= 0
			var:war_assessment >= -7000
		}
	}
	text = {
		localization_key = DIPLOMATIC_PLAY_WAR_NOT_WILLING
		trigger = {
			var:war_assessment < 7000
		}
	}
}
