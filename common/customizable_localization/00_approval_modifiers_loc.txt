	text = {
		localization_key = approval_modifier_co_ruler_party_member_loc_1

			root.character_party = employer.current_ruler.character_party 
		}
	}

	text = {
		localization_key = approval_modifier_co_ruler_party_member_loc_2

		}
	}

		trigger = {
			always = yes
		}
	}
}
