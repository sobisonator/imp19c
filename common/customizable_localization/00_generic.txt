#Correct loc for injury type.
get_stability_7_rand = {
	type = country
	random_valid = yes
	text = {
		localization_key = stability.7.rand.1
		trigger = {
			current_ruler = {
				OR = {
					has_trait = maimed
					has_trait = wounded
				}
			}
		}
	}
	text = {
		localization_key = stability.7.rand.2
		trigger = {
			current_ruler = {
				OR = {
					has_trait = epileptic
				}
			}
		}
	}
	text = {
		localization_key = stability.7.rand.3
		trigger = {
			NOT = {
				current_ruler = {
					OR = {
						has_trait = epileptic
						has_trait = maimed
						has_trait = wounded
					}
				}
			}
		}
	}
	text = {
		localization_key = stability.7.rand.1
		trigger = {
			always = yes
		}
	}
}
