					}
				}

					random_country_culture = {
						limit = {
							is_integrated = no
						}
						weight = {
							modifier = {
								factor = country_culture_pop_count

						limit = {
							is_integrated = no
						}
						weight = {
							modifier = {
								factor = country_culture_pop_count
							}
						}
						scope:civil_war_country = {
							integrate_country_culture = prev
