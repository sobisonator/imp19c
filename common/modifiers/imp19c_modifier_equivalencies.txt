# This file documents how certain vanilla modifiers are equivalent to Imperatrix modifiers

# ADMIN
# global_export_commerce_modifier = modifier to admin points required nationwide
# global_building_slot = add to admin points required nationwide
# state_commerce_modifier = modifier to admin points required by this state
# global_tax_modifier = modifier to admin points required by all states nationwide

# MILITARY & NAVY
# army_maintenance_cost = modifier to military supplies consumption by armies
# navy_maintenance_cost = modifier to naval supplies consumption
# fort_maintenance_cost = modifier to military supplies consumption by forts

# ^ loc done

# TAXATION & WEALTH GENERATION
# local_tax_modifier = modifier to wealth generated from production in province
# global_tax_modifier = modifier to wealth generated from production in all provinces
# commerce_value = add to commerce district wealth production in province
# global_commerce_modifier = modifier to all commerce districts wealth production

# GOODS PRODUCTION
# tribute_income_modifier = modifier to agriculture output in all provinces
# omen_power = modifier to mining output in all provinces
# global_import_commerce_modifier = modifier to industry output in all provinces

# DIPLOMACY
# omen_power = percentage modifier to DIPLOMACY_power

# SHIPPING & TRADE
# local_base_trade_routes = add to shipping power in province
# local_base_trade_routes_modifier = modify shipping power from infrastructure in province
# global_state_trade_routes = add to tradezone penetration from trade agreements
# global_capital_trade_routes = add to tradezone penetration from own shipping power
