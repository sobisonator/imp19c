local_autonomy_svalue = {
	value = 0
	every_state_province = {
		if = {
			limit = {
				province_unrest > 1
			}
			add = 1
		}
		if = {
			limit = {
				province_unrest > 3
			}
			add = 2
		}
		if = {
			limit = {
				province_unrest > 5
			}
			add = {
				value = province_unrest
				multiply = 0.5
			}
		}
	}
}
