DEMAND_unfulfilled_food_need_governorship = {
	# Scope: governorship
	value = DEMAND_total_food_need_governorship
	subtract = DEMAND_fulfilled_food_need_governorship
}

DEMAND_total_food_need_governorship = {
	# Scope: governorship
	add = governorship_population
	multiply = DEMAND_need_small
}

	# Scope = governorship
	# Non-food farming and hunting - provides quarter food

	add = GOODS_governorship_tobacco_produced

TRADE_get_total_food_need_governorship = {
	# Scope: governorship
	# Function: Set 
	every_tradegood_complex = {
		APPLY = TRADE_get_total_food_need_per_good
	}
}

TRADE_get_total_food_need_per_good = {
	change_to_variable = {
		name = total_food_need_satisfied
		add = $tradegood$GOODS_governorship__produced
	}
}

TRADE_reset_every_total_food_need_governorship = {
	set_variable = {
		name = total_food_need_satisfied
	}
}
