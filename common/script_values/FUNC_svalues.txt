FUNC_rough_accepted_state_population = {
	# Scope: State
	# Function: Get state population in provinces where the primary culture is dominant
	every_state_province = {
		limit = {
			dominant_province_culture = {
				is_culture = owner.culture
			}
		}
		add = total_population
	}
}

FUNC_governorship_ports_ratio = {
	# Scope: Governorship
	# Function: Get the % of country's ports located in this governorship
	value = 0
	if = {
		limit = {
			any_governorship_state = {
				any_state_province = {
					is_sea = no
					is_coastal = yes
					trigger_if = {
						limit = {
							is_coastal = yes
						}
						has_building = port_building
					}
				}
			}
		}
		every_governorship_state = {
			every_state_province = {
				add = num_of_port_building
			}
		}
		divide = {
			value = owner.FUNC_num_ports_in_country
			min = 1
		}
	}
}

FUNC_governorship_ratio_national_forts = {
	# Scope: Governorship
	# Function: Get the percentage of the country's forts that are located in this governorship
	value = 0
	if = {
		limit = {
			owner = {
				FUNC_country_total_forts > 0 
			}
		}
		every_governorship_state = {
			every_state_province = {
				limit = {
					num_of_fortress_building > 0
				}
				add = num_of_fortress_building
			}
		}
		owner = {
			divide = FUNC_country_total_forts
		}
	}
}

}
