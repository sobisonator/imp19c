GOODS_governorship_alcohol_capacity = {
	value = INDUSTRY_alcohol_factories
	multiply = 5
	# COTTAGE INDUSTRIES
	if = { # Alcohol from grain & vine
		limit = {
			OR = {
				has_variable = produces_grain
				has_variable = produces_mediterranean_fruit
			}
		}
		add = GOODS_cottage_industry_from_grain
		add = GOODS_cottage_industry_from_mediterranean_fruit
	}
}
GOODS_governorship_alcohol_produced = {
	value = GOODS_governorship_alcohol_capacity
	# DEFICIT MALUSES
	if = { # Malus from not enough grain
		limit = {
			GOODS_grain_stockpile < DEMAND_grain_from_industry_alcohol
		}
		multiply = DEMAND_grain_industry_alcohol_malus
	}
	if = { # Malus from not enough mediterranean fruit
		limit = {
			GOODS_mediterranean_fruit_stockpile < DEMAND_mediterranean_fruit_from_industry_alcohol
		}
		multiply = DEMAND_mediterranean_fruit_industry_alcohol_malus
	}
}
