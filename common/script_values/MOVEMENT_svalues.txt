#NOTES: Every trade zone has a base value for transportation between EVERY other trade node. This value should be generated with a script and based on the setup spreadsheet so there is minimal performance impact.
# Each connection between each zone has an individual svalue.
# Each zone also has a transportation value, this is how the movement cost reductions like roads and railroads are factored in. Each connecting svalue should subtract the transportation value of BOTH of the nodes it is connecting. This doesn't work perfect...ideally we would also factor in the roads in between nodes.
north_italian_transportation_svalue = {
	value = 0
	every_region = {
		limit = {
			north_italian_tradezone = {
				PROVINCE = no
			}
		}
		every_region_province = {
			limit = {
				any_neighbor_province = {
					has_road_towards = PREV
				}
			}
			add = 0.2
		}
	}
	#ADD RAILROADS HERE AS WELL, RAILROAD VALUE IS 0.5
}
south_italian_transportation_svalue = {
	value = 0
	every_region = {
		limit = {
			south_italian_tradezone = {
				PROVINCE = no
			}
		}
		every_region_province = {
			limit = {
				any_neighbor_province = {
					has_road_towards = PREV
				}
			}
			add = 0.2
		}
	}
	#ADD RAILROADS HERE AS WELL, RAILROAD VALUE IS 0.5
}
north_italian_to_south_italian_svalue = {
	value = 100 #This is the base value, all connections will need a base value. Ideally a script is written so these values can be generated.
	subtract = north_italian_transportation_svalue
	subtract = south_italian_transportation_svalue
}
