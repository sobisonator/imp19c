INDUSTRY_assign_factory = {
	# Scope: Governorship
	# Function: Change the number of factories by the specified amount
	change_variable = {
		name = INDUSTRY_factories_assigned_$tradegood$
		add = $amount$
	}
}

INDUSTRY_cache_slots = {
	# Scope: Governorship
	# Function: Cache the number of industry slots in use and available to make GUI checks cheaper
	set_variable = {
		name = INDUSTRY_governorship_total_industry_slots
		value = INDUSTRY_governorship_total_industry_slots
	}
	set_variable = {
		name = INDUSTRY_governorship_used_industry_slots
		value = INDUSTRY_governorship_used_industry_slots
	}
	set_variable = {
		name = INDUSTRY_governorship_free_industry_slots
		value = {
			var:INDUSTRY_governorship_total_industry_slots
			subtract = var:INDUSTRY_governorship_used_industry_slots
		}
	}
}
