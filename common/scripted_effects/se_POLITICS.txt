POLITICS_upset_pacifists = {
	# Scope: Country
	# Function: Upset all pacifists in the country
	custom_tooltip = upset_pacifists
    hidden_effect = {
        every_character = {
            limit = {
                charisma > martial
            }
            add_loyalty = loyalty_upset_ideology_$level$
        }  
    }
                 
}
