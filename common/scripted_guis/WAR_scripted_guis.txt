demand_reparations_button = {
	scope = war
	saved_scopes = {
		player
	}
	ai_is_valid = {
		always = no
	}
	is_valid = {
		trigger_if = {
			limit = { is_civil_war = yes }
			is_civil_war = no
		}
		trigger_if = {
			limit = { 
				any_war_defender = { this = scope:player } 
			}
			trigger_if = {
				limit = {
					NOT = { defender_warleader = scope:player }
				}
				custom_tooltip = {
					text = player_is_the_defending_leader
					defender_warleader = scope:player
				}
			}
			custom_tooltip = {
				text = has_greater_than_50_warscore_tt
			    war_score = {
			        target = defender_warleader
			        value >= 50
			    }
		    }
		}
		trigger_if = {
			limit = { 
				any_war_attacker = { this = scope:player } 
			}
			trigger_if = {
				limit = {
					NOT = { attacker_warleader = scope:player }
				}
				custom_tooltip = {
					text = player_is_the_attacking_leader
					attacker_warleader = scope:player
				}
			}
			custom_tooltip = {
				text = has_greater_than_50_warscore_tt
			    war_score = {
			        target = attacker_warleader
			        value >= 50
			    }
		    }
		}
	}
	effect = {
		save_scope_as = war
		if = {
			limit = {
				NOT = { has_variable = current_warscore_cost_war }
			}
			set_variable = {
				name = current_warscore_cost_war
				value = 50
			}
		}
		else_if = {
			limit = {
				has_variable = current_warscore_cost_war
			}
			change_variable = {
				name = current_warscore_cost_war
				add = 50
			}
		}
		scope:player = {
			set_variable = demanding_war_reparations
			if = {
				limit = {
					NOT = { has_variable = current_warscore_cost }
				}
				set_variable = {
					name = current_warscore_cost
					value = 50
				}
			}
			else_if = {
				limit = {
					has_variable = current_warscore_cost
				}
				change_variable = {
					name = current_warscore_cost
					add = 50
				}
			}
			if = {
				limit = {
					NOT = { has_variable = aggressive_expansion_cost }
				}
				set_variable = {
					name = aggressive_expansion_cost
					value = 10
				}
			}
			else_if = {
				limit = {
					has_variable = aggressive_expansion_cost
				}
				change_variable = {
					name = aggressive_expansion_cost
					add = 10
				}
			}
			calculate_ai_acceptance_effect = yes
		}
	}	
}
demand_reparations_button_cancel = {
	scope = war
	saved_scopes = {
		player
	}
	effect = {
		change_variable = {
			name = current_warscore_cost_war
			subtract = 50
		}			
		scope:player = {
			remove_variable = demanding_war_reparations
			change_variable = {
				name = aggressive_expansion_cost
				subtract = 10
			}
			change_variable = {
				name = current_warscore_cost
				subtract = 50
			}
			calculate_ai_acceptance_effect = yes
		}
	}
	
}

alt_peace_button = {
	scope = war

	saved_scopes = {
		player
	}
	ai_is_valid = {
		always = no
	}
	effect = {
		save_scope_as = peace_root
		every_war_attacker = {
			limit = {
				NOT = { THIS = root.attacker_warleader }
			}
			scope:player = {
				add_to_variable_list = {
					name = war_attacker_list
					target = prev
				}
			}
			scope:peace_root = {
				add_to_variable_list = {
					name = war_attacker_list
					target = prev
				}
			}			
		}
		every_war_defender = {
			limit = {
				NOT = { THIS = root.defender_warleader }
			}
			scope:player = {
				add_to_variable_list = {
					name = war_defender_list
					target = prev
				}
			}
			scope:peace_root = {
				add_to_variable_list = {
					name = war_defender_list
					target = prev
				}
			}
		}
		if = {
			limit = {
				any_war_attacker = { this = scope:player } 
			}
			scope:player = {
				set_variable = {
					name = bad_guys_var
					value = root.defender_warleader
				}
			}
		}
		if = {
			limit = {
				any_war_defender = { this = scope:player } 
			}
			scope:player = {
				set_variable = {
					name = bad_guys_var
					value = root.attacker_warleader
				}
			}
		}

		# Make this a scripted effect to easily double it up for defender and attacker

		# Lists of participating countries
		clear_variable_list = defender_tags
		clear_variable_list = attacker_tags

		# Defender
		clear_variable_list = provinces_available_from_defender
		clear_variable_list = states_available_from_defender
		clear_variable_list = governorships_available_from_defender
		# Attacker
		clear_variable_list = provinces_available_from_attacker
		# Both
		clear_variable_list = provinces_available_from_both

		every_war_defender = {
			save_scope_as = target_defender
			clear_variable_list = states_available_from_defender
			every_governorships = {
				save_scope_as = this_governorship
				clear_variable_list = states_available_from_defender
				scope:peace_root = {
					add_to_variable_list = {
						name = governorships_available_from_defender
						target = scope:this_governorship
					}
				}
				every_governorship_state = {
					save_scope_as = this_state
					clear_variable_list = provinces_available_from_defender
					scope:this_governorship = {
						add_to_variable_list = {
							name = states_available_from_defender
							target = scope:this_state
						}
					}
					scope:target_defender = {
						add_to_variable_list = {
							name = states_available_from_defender
							target = scope:this_state
						}						
					}
					every_state_province = {
						save_scope_as = this_province
						scope:this_state = {
							add_to_variable_list = {
								name = provinces_available_from_defender
								target = scope:this_province
							}
						}
						scope:peace_root = {
							add_to_variable_list = {
								name = provinces_available_from_both
								target = scope:this_province
							}
						}
					}
				}
			}
		}

		every_war_attacker = {
			every_owned_province = {
				save_scope_as = this_province
				scope:peace_root = {
					add_to_variable_list = {
						name = provinces_available_from_attacker
						target = scope:this_province
					}
				}
			}		
		}

	}
}

send_final_offer_button = { #This runs the effects and checks if Ai will accept.
	scope = war
	saved_scopes = {
		player
	}
	is_valid = {
		hidden:scope:player = {
			trigger_if = {
				limit = {
					root = {
						war_score = {
							target = scope:player
							value > 0
						}
					}
				}
				custom_tooltip = {
					text = has_greater_than_0_warscore_tt
					root = {
						war_score = {
							target = scope:player
							value > 0
						}
					}
				}
			}
			trigger_else = {
				trigger_if = {
					limit = {
						has_variable = ai_will_accept
					}
					custom_tooltip = {
						text = ai_does_accept_this_offer_tt
						has_variable = ai_will_accept
					}
				}
				trigger_else_if = {
					limit = {
						has_variable = ai_will_not_accept
					}
						custom_tooltip = {
							text = ai_does_not_accept_this_offer_tt
							NOT = { has_variable = ai_will_not_accept }
						}
				}
				trigger_else = {
					limit = {
						NOT = { has_variable = ai_acceptance_var }
					}
					custom_tooltip = {
						text = has_not_sent_an_offer_yet_tt
						has_variable = ai_acceptance_var
					}
				}
			}
		}
	}
	effect = {
		scope:player = {
			if = {
				limit = {
					has_variable = current_warscore_cost
				}
				peace_out_of_war_effect = yes
			}
			if = {
				limit = { has_variable = aggressive_expansion_cost }
				add_aggressive_expansion = {
					value = var:aggressive_expansion_cost
				}
			}
			if = {
				limit = { has_variable = extort_population_peace_var }
				var:bad_guys_var.capital_scope = {
					custom_tooltip = extort_enemy_pops_tt
					hidden_effect = {
						ordered_pops_in_province = {
							order_by = pop_hapiness
							max = {
								integer_range = {
									max = 10
									min = 5
								}
							}
							move_pop = scope:player.capital_scope
						}
					}
				}
			}
			if = {
				limit = { has_variable = steal_knowledge_peace_treaty_var }
				add_innovation = 1
				add_research = {
					technology = civic_tech
					value = 5
				}
				add_research = {
					technology = oratory_tech
					value = 5
				}
				add_research = {
					technology = military_tech
					value = 5
				}
				add_research = {
					technology = religious_tech
					value = 5
				}
			}
			if = {
				limit = {
					has_variable = demands_gold
				}
				add_treasury = {
					value = var:demands_gold
					multiply = 1
				}
				var:bad_guys_var = {
					add_treasury = {
						value = 0
						subtract = scope:player.var:demands_gold
					}
				}
			}
			if = {
				limit = {
					has_variable = destabilize_peace_treaty_var
				}
				var:bad_guys_var = {
					add_stability = -15
					add_political_influence = -35
					current_ruler = {
						add_popularity = -10
					}
				}
			}
			if = {
				limit = {
					has_variable = delegitimize_peace_treaty_var
				}
				var:bad_guys_var = {
					add_country_modifier = {
						name = delegitimized_administration
						duration = 3650
					}
					add_political_influence = -35
					add_legitimacy = -20
					current_ruler.family = {
						add_prestige = -250
					}
				}
			}
		}
		if = {
			limit = {
				has_variable = demanding_war_reparations
			}
			if = {
				limit = {
					any_war_defender = { this = scope:player }
				}
				attacker_warleader = { save_scope_as = attacker }
				custom_tooltip = 10_percent_income_for_10_years_defender_tt
				attacker_warleader = {
					set_variable = {
						name = paying_5_year_reparations
						days = 1825
						value = scope:player 
						#Value is scope:player so this variable can be used to check if money should be given and give the money at the same time.	 
						#This also makes it so this will work in multiplayer which is nice
					}
				}
			}
			if = {
				limit = {
					any_war_attacker = { this = scope:player }
				}
				defender_warleader = { save_scope_as = defender }
				custom_tooltip = 10_percent_income_for_10_years_attacker_tt
				defender_warleader = {
					set_variable = {
						name = paying_5_year_reparations
						days = 1825
						value = scope:player
					}
				}
			}
		}
		hidden_effect = { force_white_peace = THIS }
	}
}
