add_clothing_button = {
    scope = governorship
    saved_scopes = { player }
    is_valid = {
        #Tech Triggers
        trigger_if = {
            limit = {
                NOT = { scope:player = { can_pay_price = add_industry_price } }
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
            }
            hidden:scope:player = { can_pay_price = add_industry_price }
        }
        trigger_else = {
            limit = {
                NOT = { scope:player = { can_pay_price = remove_industry_price } }
            }
            hidden:scope:player = { can_pay_price = remove_industry_price }
        }
    }
    effect = {
        save_scope_as = governorship_name
        if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
                has_variable = INDUSTRY_factories_assigned_clothing
            }
            scope:player = { pay_price = add_industry_price }
            change_variable = {
                name = INDUSTRY_factories_assigned_clothing
                add = 1
            }
            custom_tooltip = clothing_add_button_tt
        }
    }
}
remove_clothing_button = {
    scope = governorship
    effect = {
        if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
            }
            custom_tooltip = two_newlines_tooltip
        }
        save_scope_as = governorship_name
        show_as_tooltip = { scope:player = { pay_price = remove_industry_price } }
        if = {
            limit = {
                var:INDUSTRY_factories_assigned_clothing > 0
                scope:player = { can_pay_price = remove_industry_price }
            }
            hidden_effect = { scope:player = { pay_price = remove_industry_price } }
            change_variable = {
                name = INDUSTRY_factories_assigned_clothing
                subtract = 1
            }
        }
        custom_tooltip = clothing_remove_button_tt
    }
}
