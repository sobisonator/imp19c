add_clothing_button = {
    scope = governorship
    saved_scopes = { player }
    is_valid = {
        trigger_if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
            }
            custom_tooltip = {
                text = industries_at_maximum_tt
            }
        }
        trigger_if = {
            limit = {
                scope:player = {
                    NOT = { can_pay_price = add_industry_price }
                }
            }
            hidden:scope:player = { can_pay_price = add_industry_price }
        }
    }
    effect = {
        scope:player = {
            pay_price = add_industry_price
        }
        save_scope_as = governorship_name
        custom_tooltip = clothing_button_tt
        if = {
            limit = {
                has_variable = INDUSTRY_factories_assigned_clothing
            }
            change_variable = {
                name = INDUSTRY_factories_assigned_clothing
                add = 1
            }
        }
    }
}
remove_clothing_button = {
    scope = governorship
    effect = {
        if = {
            limit = {
                var:INDUSTRY_factories_assigned_clothing > 0
            }
            change_variable = {
                name = INDUSTRY_factories_assigned_clothing
                subtract = 1
            }
        }
    }
}

    saved_scopes = { player }
    is_valid = {
        trigger_if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
            }
            custom_tooltip = {
                text = industries_at_maximum_tt
            }
        }
        trigger_if = {
            limit = {
                scope:player = {
                    NOT = { can_pay_price = add_industry_price }
                }
            }
            hidden:scope:player = { can_pay_price = add_industry_price }
        }
    }
    effect = {
        scope:player = {
            pay_price = add_industry_price
        }
        save_scope_as = governorship_name
        custom_tooltip = late_artillery_button_tt
        if = {
            limit = {
                has_variable = INDUSTRY_factories_assigned_late_artillery
            }
            change_variable = {
                name = INDUSTRY_factories_assigned_late_artillery
                add = 1
            }
        }
    }
}
remove_late_artillery_button = {
    scope = governorship
    effect = {
        if = {
            limit = {
                var:INDUSTRY_factories_assigned_late_artillery > 0
            }
            change_variable = {
                name = INDUSTRY_factories_assigned_late_artillery
                subtract = 1
            }
        }
    }
}
