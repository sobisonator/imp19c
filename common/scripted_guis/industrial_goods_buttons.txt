add_clothing_button = {
    scope = governorship
    saved_scopes = { player }
    confirm_title = "clothing_industry_confirm_title"
    confirm_text = "clothing_industry_confirm_desc"

    is_valid = {
        
        owner = { invention = tech_manufactories }
    	owner = { invention = tech_cotton_gin }

        
        #Tech Triggers
    }
    effect = {
        save_scope_as = governorship_name
        if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
                has_variable = INDUSTRY_factories_assigned_clothing
            }
            scope:player = { pay_price = add_industry_price }
            INDUSTRY_assign_factory = {
                tradegood = clothing
                amount = 1
            }
            custom_tooltip = clothing_add_button_tt
        }
    }
}
remove_clothing_button = {
    scope = governorship
    confirm_title = "clothing_industry_remove_confirm_title"
    confirm_text = "clothing_industry_remove_confirm_desc"

    effect = {
        if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
            }
            custom_tooltip = two_newlines_tooltip
        }
        save_scope_as = governorship_name
        show_as_tooltip = { scope:player = { pay_price = remove_industry_price } }
        if = {
            limit = {
                var:INDUSTRY_factories_assigned_clothing > 0
                scope:player = { can_pay_price = remove_industry_price }
            }
            hidden_effect = { scope:player = { pay_price = remove_industry_price } }
            INDUSTRY_assign_factory = {
                tradegood = clothing
                amount = -1
            }
        }
        custom_tooltip = clothing_remove_button_tt
    }
}

    confirm_title = "wooden_ships_industry_remove_confirm_title"
    confirm_text = "wooden_ships_industry_remove_confirm_desc"

    effect = {
        if = {
            limit = {
                INDUSTRY_governorship_total_industry_slots > INDUSTRY_governorship_used_industry_slots
            }
            custom_tooltip = two_newlines_tooltip
        }
        save_scope_as = governorship_name
        show_as_tooltip = { scope:player = { pay_price = remove_industry_price } }
        if = {
            limit = {
                var:INDUSTRY_factories_assigned_wooden_ships > 0
                scope:player = { can_pay_price = remove_industry_price }
            }
            hidden_effect = { scope:player = { pay_price = remove_industry_price } }
            INDUSTRY_assign_factory = {
                tradegood = wooden_ships
                amount = -1
            }
        }
        custom_tooltip = wooden_ships_remove_button_tt
    }
}
