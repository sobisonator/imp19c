#Trade zone testing
#
#$PROVINCE$ refers to the scope it is used in, region or province can be either yes or no it just depends on how it is being used.
south_italian_tradezone = {
	trigger_if = {
		limit = {
			$PROVINCE$ = yes
		}
		OR = {
			is_in_region = Sicily
			is_in_region = Central_Italy
			is_in_region = Southern_Italy
		}
	}
	trigger_else = {
		limit = {
			$PROVINCE$ = no
		}
		OR = {
			this = region:Sicily
			this = region:Central_Italy
			this = region:Southern_Italy
		}
	}
}
