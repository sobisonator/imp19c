DIPLOMACY_play_war_willing_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator is willing to go to war
	var:war_assessment > DIPLOMACY_play_war_willing
}
DIPLOMACY_play_war_considering_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator is considering going to war
	var:war_assessment <= DIPLOMACY_play_war_willing
	var:war_assessment >= DIPLOMACY_play_war_considering
}
DIPLOMACY_play_war_not_willing_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator is not willing to go to war
	var:war_assessment < DIPLOMACY_play_war_considering
}

DIPLOMACY_play_attitude_friendly_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator is friendly towards target
	var:AI_play_attitude > DIPLOMACY_play_attitude_friendly
}
DIPLOMACY_play_attitude_hostile_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator is hostile towards target
	var:AI_play_attitude < DIPLOMACY_play_attitude_hostile
}
DIPLOMACY_play_attitude_neutral_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator is hostile towards target
	var:AI_play_attitude >= DIPLOMACY_play_attitude_hostile
	var:AI_play_attitude <= DIPLOMACY_play_attitude_friendly
}

DIPLOMACY_play_colonial_trigger = {
	# Scope: Province (diplomatic play)
	# Instigator capital is significantly more industrialised than target area
	DIPLOMACY_play_industrialisation_diff > DIPLOMACY_play_colonial_diff
}
