country_decisions = {
	establish_public_debt_administration = {
		potential = {
			has_variable = official_currency
			NOT = {
				has_variable = public_debt_administration
			}
		}
		allow = {
			invention = tech_public_debt_administration
			political_influence > 10
		}
		effect = {
			set_variable = public_debt_administration
			add_political_influence = -10
		}
	}

	adopt_currency = {
		potential = {
			always = no # WiP
			NOT = {
				has_variable = official_currency
			}
		}
		allow = {
			OR = {
				# Any neighbour or ally has a currency
				any_country = {
					has_variable = official_currency
					OR = {
						alliance_with = ROOT
						any_owned_province = {
							any_neighbor_province = {
								owner = ROOT
							}
						}
					}
				}
			}
		}
		effect = {
			trigger_event = {
				id = adopt_currency.1 # Pick which currency out of 4 from allies or neighbours, ordered by diplomacy power of the ally or neighbour
			}
		}
	}
}
