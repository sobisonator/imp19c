country_decisions = {
	invite_bolivar = {
		potential = {
			char:59 = { # Bolivar
				is_alive = yes
			}
			any_country = {
				war_with = ROOT
				OR = {
					tag = SPA
					tag = CHA
					trigger_if = {
						limit = {
							is_subject = yes
						}
						OR = {
							overlord = c:SPA
							overlord = c:NSP
						}
					}
				}
			}
		}
		allow = {
			war = yes
			government = revolutionary_republic
			political_influence > 50
			NOT = {
				has_variable = invited_bolivar
			}
			char:59 = {
				NOT = {
					has_variable = bolivar_invited_recently
					employer = ROOT
				}
			}
		}
		effect = {
			set_variable = invited_bolivar
			add_political_influence = -50
			char:59 = {
				set_variable = {
					name = bolivar_invited_recently
					days = 1095 # 3 Years
				}
			}
		}
	}
}
