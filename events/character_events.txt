character_events.4 = {
	type = character_event
	hidden = yes
	
	immediate = {
		if = {
			limit = {
				is_governor = no
				is_mercenary = no
				trigger_if = {
					limit = {
						has_variable = came_from
					}
					NOT = { employer = { civil_war_with = root.var:came_from } }
				}
				NOT = { has_character_modifier = foreign_citizen }
			}
			add_character_modifier = {
				name = foreign_citizen
				duration = -1
			}
		}
	}

}
