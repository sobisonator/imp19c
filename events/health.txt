namespace = health

health.1 = { #Death during childbirth
	type = character_event
	title = "health.1.t"
	desc = "health.1.desc"
	picture = greek_court
	
	left_portrait = root
	right_portrait = scope:grieving_husband
	
	trigger = {
		is_alive = yes
		exists = employer 
	}
	
	
	immediate = {
		scope:newborn = {
			save_scope_as = newborn
		}
		if = {
			limit = {
				has_trait = infection
			}
			death = {
				death_reason = death_infection_during_childbirth
			}
		}
		else_if = {
			limit = {
				NOT = {
					has_trait = infection
				}
			}
			death = {
				death_reason = death_childbirth
			}
		}
		employer = {
			if = {
				limit = {
					any_character = {
						is_spouse_of = root
						is_alive = yes
					}
				}
				random_character = {
					limit = {
						is_spouse_of = root
					}
					save_scope_as = grieving_husband
				}
			}
		}
	}
	
	option = {
		name = "health.1.a"
		if = {
			limit = {
				exists = scope:grieving_husband
				scope:grieving_husband = {
					NOT = {
						has_trait = mourning
					}
				}
			}
			random = {
			chance = 50
				scope:grieving_husband = {
					add_trait = mourning
					trigger_event = {
						id = health.2
						days = { 360 3600 }
					}
				}
			}	
		}
	}
	
}
