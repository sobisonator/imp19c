ECON_events.1 = {	#Trigger this event to test trade centers.
    type = country_event
    hidden = yes

    trigger = {
        NOT = {
            has_global_variable = economic_enhancement_setup_done
        }
    }

    immediate = {
        #Populate the world with trade centers - 1 per region.
        set_global_variable = economic_enhancement_setup_done
        tradezone_setup_effect = {
            TRADE_ZONE = north_italian_tradezone
        }
        tradezone_setup_effect = {
            TRADE_ZONE = south_italian_tradezone
        }
    }
}
