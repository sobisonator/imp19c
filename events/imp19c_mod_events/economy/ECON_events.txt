namespace = ECON_events
ECON_events.1 = {	#Trigger this event to test trade centers.
	#NOTE: This first event should NOT exist this should all be moved to setup this is just nice for testing.
	#IF you do try to move it to setup first test if provinces modifiers that are set in setup are removed on conquest(im 99% sure they don't but should look)
    type = country_event
    hidden = yes

    trigger = {
        NOT = {
            has_global_variable = economic_enhancement_setup_done
        }
    }

    immediate = {
    	#Old code from FMO, here for trade zone testing.
        #Populate the world with trade centers - 2 per region.
        set_global_variable = economic_enhancement_setup_done
        every_region = {
            if = {
                limit = {
                    #Do this so there are not more than 4 trade centers in a region at game start.
                    #90% of regions should have 3 centers but some will have 4 at game start.
                    any_trade_center = {
                        count < 1
                    }
                }
                ordered_region_province = {
                    limit = {
                        NOR = {
                            has_province_modifier = regional_center_of_trade_level_2
                            has_province_modifier = regional_center_of_trade_level_3
                        }
                    }
                    order_by = total_population
                    max = 2
                    add_permanent_province_modifier = {
                        name = regional_center_of_trade_level_1
                    }
                }
            }
        }
    }
}
