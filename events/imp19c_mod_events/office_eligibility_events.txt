namespace = office_eligibility


office_eligibility.1 = {
	type = country_event
	hidden = yes
	immediate = {
			every_character = {
				if = {
					limit = {
						employer = { tag = CHI }
						OR = {
							has_trait = jinshi
							AND = {
								has_trait = gongshi
								OR = {
									finesse >= 8
									charisma >= 9
									zeal >= 10
									martial >= 9
								}
							}
							AND = {
								OR = {
									has_culture = manchu
									has_culture = mongolian
								}
								OR = {
									prominence >= 70
									martial >= 9
								}
							}
						}
						from_ruler_family = no
					}
					add_triggered_character_modifier = {
						name = EligibleForOffice
						duration = -1
					}
				}
				else_if = {
					limit = {
						employer = { tag = CHI }
						OR = {
							has_trait = gongshi
							has_trait = juren
							AND = {
								OR = {
									finesse >= 10
									charisma >= 11
									zeal >= 12
									martial >= 13
								}
								OR = {
									AND = {
										has_trait = shengyuan
										OR = {
											finesse >= 9
											charisma >= 10
											zeal >= 11
											martial >= 12
										}
									}
									AND = {
										OR = {
											has_culture = manchu
											has_culture = mongolian
										}
										OR = {
											martial >= 8
											prominence >= 50
										}
									}
								}
							}
						}
						from_ruler_family = no
					}
					add_triggered_character_modifier = {
						name = EligibleForJobs
						duration = -1
					}
				}
				else_if = {
					limit = {
						employer = { tag = CHI }
						OR = {
							martial >= 7
							prominence >= 55
						}
					}
					add_triggered_character_modifier = {
						name = EligibleForCommandOnly
						duration = -1
					}
				}
				else_if  = {
					limit = {
						OR = {
							AND = {
								employer = { is_monarchy = yes }
								from_ruler_family = yes 
								employer = {
									has_variable = monarch_character
								}
							}
							trigger_if = {
								limit = {
									employer = { has_variable = monarch_character }
								}
								has_same_family = employer.var:monarch_character
							}
						}
					}
					add_triggered_character_modifier = {
						name = IneligibleForAllRoles
						duration = -1
					} 
				}
				else = {
					add_gold = 1
				}
			}
			every_character = {
				if = {
					limit = {
						OR = {
							AND = {
								employer = { is_monarchy = yes }
								from_ruler_family = yes 
								employer = {
									has_variable = monarch_character
								}
							}
							trigger_if = {
								limit = {
									employer = { has_variable = monarch_character }
								}
								has_same_family = employer.var:monarch_character
							}
						}
						has_any_office = yes
					}
					remove_all_offices = yes
				}
				else_if = {
					limit = {
						OR = {
							AND = {
								employer = { is_monarchy = yes }
								from_ruler_family = yes 
								employer = {
									has_variable = monarch_character
								}
							}
							trigger_if = {
								limit = {
									employer = { has_variable = monarch_character }
								}
								has_same_family = employer.var:monarch_character
							}
						}
						has_tech_office = yes
					}
					remove_as_researcher = yes
				}
				else_if = {
					limit = {
						OR = {
							AND = {
								employer = { is_monarchy = yes }
								from_ruler_family = yes 
								employer = {
									has_variable = monarch_character
								}
							}
							trigger_if = {
								limit = {
									employer = { has_variable = monarch_character }
								}
								has_same_family = employer.var:monarch_character
							}
						}
						is_governor = yes
					}
					remove_as_governor = yes
				}
				else_if = {
					limit = {
						employer = { tag = CHI }
						NOR = {
							has_trait = jinshi
							has_trait = gongshi
							has_culture = manchu
							has_culture = mongolian
						}
						has_any_office = yes
					}
					remove_all_offices = yes
				}
				else_if = {
					limit = {
						employer = { tag = CHI }
						NOR = {
							has_trait = jinshi
							has_trait = gongshi
							has_trait = shengyuan
							has_culture = manchu
							has_culture = mongolian
							martial = 12
						}
						is_governor = yes
					}
					remove_as_governor = yes
				}
				else_if  = {
					limit = {
						NOR = {
							has_trait = jinshi
							has_trait = gongshi
							has_trait = shengyuan
							has_culture = manchu
							has_culture = mongolian
							martial = 10
							finesse = 10
							charisma = 10
							zeal = 10
						}
						has_tech_office = yes
					}
					remove_as_researcher = yes
				}
				else = {
					add_gold = 1
				}
			}
		}
}
